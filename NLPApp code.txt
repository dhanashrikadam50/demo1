class NLPApp:
    def __init__(self):
        self.database = {}
        self.first_menu()

    def first_menu(self):
        while True:
            choice = input('''Hi! How would you like to proceed?
    1. Not a member? Register
    2. Already Member? Login
    3. Exit
    ''')
            if choice == '1':
                self.register()
            elif choice == '2':
                self.login()
            else:
                print("Invalid choice. Please enter 1 OR 2.")

    def register(self):
        print("Register")
        name = input("Enter Name: ")
        email = input("Enter Email: ")
        password = input("Enter Password: ")

        if email in self.database:
            print("The email already exists")
        else:
            self.database[email] = [name, password]
            print("Registration successful. Please login.")

    def login(self):
        print("Login")
        email = input("Enter Email: ")
        password = input("Enter Password: ")

        if email in self.database and self.database[email][1] == password:
            print(f"Login successful. Welcome, {self.database[email][0]}!")
        else:
            print("Invalid email or password")
